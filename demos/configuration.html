});

☓

Custom Animations

Please check out the documentation of Morpheus for animation information. The animationIn and animationOut objects are passed directly into Morpheus.

☓ 
$("#overlay-animation").overlay({
    trigger: "#trigger-animation",
    position: "fixed",

    startAnimationCss: {
        opacity: 0,
        top: 0
    },
    animationIn: {
        opacity: 1,
        top: 80,
        duration: 300
    },
    animationOut: {
        opacity: 0,
        top: -200,
        duration: 300
    }
});
